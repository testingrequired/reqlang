28struct Backend {

30    file_texts: Mutex<HashMap<Url, String>>,
31}

33impl Backend {
34    pub fn new(client: Client) -> Self {
35        Self {
36            client,
37            file_texts: Mutex::new(HashMap::new()),
38        }
39    }
40
41    async fn parse_file_for_client(&self, uri: &Url, source: &str) {
42        let ast = Ast::from(source);
43
44        let result = match parse(&ast) {
45            Ok(parsed_request_file) => {
46                // Clear diagnostics on the client
47                self.client
48                    .publish_diagnostics(uri.clone(), vec![], None)
49                    .await;
50
51                Ok(parsed_request_file.into())
52            }
53            Err(errs) => {
54                let diagnostics = get_diagnostics(&errs, source);
55
56                // Log error diagnostics to client
57                // This is mostly for debugging
58                self.client
59                    .log_message(
60                        MessageType::ERROR,
61                        format!(
62                            "{} errors parsing file '{uri}':\n{diagnostics:#?}",
63                            diagnostics.len()
64                        ),
65                    )
66                    .await;
67
68                // Send error diagnostics to client
69                self.client
70                    .publish_diagnostics(
71                        uri.clone(),
72                        diagnostics
73                            .iter()
74                            .map(|x| (LspDiagnosis((*x).clone())).into())
75                            .collect(),
76                        None,
77                    )
78                    .await;
79
80                Err(errs)
81            }
82        };
83
84        // Send a notification to the client with the results of the parse
85        self.client
86            .send_notification::<ParseNotification>(ParseNotificationParams::new(
87                uri.clone(),
88                result,
89            ))
90            .await;
91    }
92}

116                text_document_sync: Some(

120                            change: TextDocumentSyncKind::FULL.into(),
121                            save: Some(
122                                SaveOptions {

124                                }
125                                .into(),
126                            ),
127                            ..Default::default()
128                        },
129                    ),
130                ),
131                ..Default::default()
132            },
133            server_info: Some(ServerInfo {
134                name: env!("CARGO_PKG_NAME").to_string(),
135                version: Some(version),
136            }),
137        })
138    }
139
140    async fn shutdown(&self) -> RpcResult<()> {
141        Ok(())
142    }
143
144    async fn did_open(&self, params: DidOpenTextDocumentParams) {
145        let uri = params.text_document.uri;
146        let source = params.text_document.text;
147
148        self.parse_file_for_client(&uri, &source).await;
149
150        let mut file_texts = self.file_texts.lock().await;
151        file_texts.insert(uri.clone(), source.clone());
152    }
153
154    async fn did_change(&self, params: DidChangeTextDocumentParams) {
155        let uri = params.text_document.uri;
156        let source = &params.content_changes.first().unwrap().text;
157
158        self.parse_file_for_client(&uri, source).await;
159    }
160
161    async fn did_save(&self, params: DidSaveTextDocumentParams) {
162        let uri = params.text_document.uri;
163        let source = params.text.unwrap_or_default();
164
165        self.parse_file_for_client(&uri, &source).await;
166
167        let mut file_texts = self.file_texts.lock().await;
168        file_texts.insert(uri.clone(), source.clone());
169    }

171    async fn execute_command(&self, params: ExecuteCommandParams) -> RpcResult<Option<Value>> {
172        self.client
173            .log_message(
174                MessageType::INFO,
175                format!("Executing command: {}", params.command.as_str()),
176            )
177            .await;
178
179        if params.command.as_str() == "reqlang.executeRequest" {
180            let from_client_params_value = params.arguments.first().expect("Should be present");
181
182            self.client
183                .log_message(MessageType::INFO, format!("{from_client_params_value:?}"))
184                .await;
185
186            // Get parsed params from JSON `Value`
187            let from_client_params =
188                Into::<RequestParamsFromClient>::into(from_client_params_value.clone());
189
190            let reqfile = template(
191                &from_client_params.reqfile,
192                from_client_params.env.as_deref(),
193                &from_client_params.prompts,
194                &from_client_params.secrets,
195                &Default::default(),
196            )
197            .expect("Should get templated request file");
198
199            let response = Into::<HttpRequestFetcher>::into(from_client_params)
200                .fetch()
201                .await
202                .expect("Request should have succeeded");
203
204            self.client
205                .log_message(
206                    MessageType::WARNING,
207                    format!("Expected response:\n{:?}", reqfile.response),
208                )
209                .await;
210
211            self.client
212                .log_message(
213                    MessageType::WARNING,
214                    format!("Actual response:\n{response:?}"),
215                )
216                .await;
217
218            if let Some(expected_response) = reqfile.response {
219                if let Err(diffs) = assert_response(&expected_response, &response) {
220                    self.client
221                        .log_message(
222                            MessageType::WARNING,
223                            format!("Differences found in the response:\n{diffs}"),
224                        )
225                        .await;
226                }
227            };
228
229            return Ok(Some(
230                serde_json::to_string(&response)
231                    .expect("Should serialize to json")
232                    .into(),
233            ));
234        };
235
236        if params.command.as_str() == "reqlang.exportRequest" {
237            let from_client_params_value = params.arguments.first().expect("Should be present");
238
239            self.client
240                .log_message(MessageType::INFO, format!("{from_client_params_value:?}"))
241                .await;
242
243            // Get parsed params from JSON `Value`
244            let from_client_params: FromClientExportRequestParams =
245                from_client_params_value.clone().into();
246
247            // Setup provider values
248            let env = from_client_params.env.as_deref();
249            let mut provider = HashMap::new();
250
251            provider.insert("env".to_string(), env.unwrap_or_default().to_string());
252
253            // Get reqfile text content
254            let url = Url::parse(&from_client_params.uri).expect("Should be a valid url");
255            let file_texts = self.file_texts.lock().await;
256            let text = file_texts.get(&url).expect("Should be present");
257
258            // Template the reqfile
259            let templated_reqfile = template(
260                text,
261                env,
262                &from_client_params.prompts,
263                &from_client_params.secrets,
264                &provider,
265            )
266            .expect("Should have templated");
267
268            let exported = export(&templated_reqfile.request, from_client_params.format);
269
270            return Ok(Some(exported.into()));
271        }
272
273        self.client
274            .log_message(MessageType::INFO, "command executed!")
275            .await;
276
277        Ok(Some(Value::String("DONE!".to_string())))
278    }

281/// Command parameters from client to export request
282///
283/// This is useful for language server clients
284#[derive(Debug, Deserialize, Serialize, Default)]
285struct FromClientExportRequestParams {
286    uri: String,
287    env: Option<String>,
288    vars: HashMap<String, String>,
289    prompts: HashMap<String, String>,
290    secrets: HashMap<String, String>,
291    format: RequestFormat,
292}
293
294impl From<Value> for FromClientExportRequestParams {
295    fn from(params_value: Value) -> Self {
296        let uri = params_value
297            .get("uri")
298            .expect("Should be present")
299            .as_str()
300            .expect("Should be a string")
301            .to_string();
302
303        let env = params_value
304            .get("env")
305            .expect("Should be present")
306            .as_str()
307            .map(|x| x.to_string());
308
309        let vars_from_params = params_value
310            .get("vars")
311            .map(|v| v.as_object().expect("Should be there"))
312            .expect("Should be there");
313
314        let mut vars: HashMap<String, String> = HashMap::default();
315
316        for (key, value) in vars_from_params {
317            vars.insert(
318                key.to_string(),
319                value.as_str().expect("Should be a string").to_string(),
320            );
321        }
322
323        let prompts_from_params = params_value
324            .get("prompts")
325            .map(|v| v.as_object().expect("Should be there"))
326            .expect("Should be there");
327
328        let mut prompts: HashMap<String, String> = HashMap::default();
329
330        for (key, value) in prompts_from_params {
331            prompts.insert(
332                key.to_string(),
333                value.as_str().expect("Should be a string").to_string(),
334            );
335        }
336
337        let secrets_from_params = params_value
338            .get("secrets")
339            .map(|v| v.as_object().expect("Should be there"))
340            .expect("Should be there");
341
342        let mut secrets: HashMap<String, String> = HashMap::default();
343
344        for (key, value) in secrets_from_params {
345            secrets.insert(
346                key.to_string(),
347                value.as_str().expect("Should be a string").to_string(),
348            );
349        }
350
351        FromClientExportRequestParams {
352            uri,
353            env,
354            vars,
355            prompts,
356            secrets,
357            format: Default::default(),
358        }
359    }
360}

391struct LspDiagnosis(Diagnosis);
392impl Deref for LspDiagnosis {
393    type Target = Diagnosis;
394
395    fn deref(&self) -> &Self::Target {
396        &self.0
397    }
398}
399impl From<LspDiagnosis> for Diagnostic {
400    fn from(value: LspDiagnosis) -> Self {
401        let range: LspDiagnosisRange = value.range.into();
402
403        Diagnostic {
404            range: range.into(),
405            severity: value.severity.map(|x| LspDiagnosisSeverity(x).into()),
406            message: value.message.clone(),
407            ..Default::default()
408        }
409    }
410}

459struct LspDiagnosisSeverity(DiagnosisSeverity);
460impl Deref for LspDiagnosisSeverity {
461    type Target = DiagnosisSeverity;
462
463    fn deref(&self) -> &Self::Target {
464        &self.0
465    }
466}
467impl From<LspDiagnosisSeverity> for DiagnosticSeverity {
468    fn from(value: LspDiagnosisSeverity) -> Self {
469        match value.0 {
470            DiagnosisSeverity::ERROR => DiagnosticSeverity::ERROR,
471            DiagnosisSeverity::HINT => DiagnosticSeverity::HINT,
472            DiagnosisSeverity::INFORMATION => DiagnosticSeverity::INFORMATION,
473            DiagnosisSeverity::WARNING => DiagnosticSeverity::WARNING,
474            _ => panic!("Invalid diagnosis severity {:?}", value.0),
475        }
476    }
477}
478impl From<DiagnosisSeverity> for LspDiagnosisSeverity {
479    fn from(value: DiagnosisSeverity) -> Self {
480        LspDiagnosisSeverity(value)
481    }

484fn serve() -> Result<()> {
485    tokio::runtime::Builder::new_current_thread()
486        .enable_all()
487        .build()
488        .context("failed to start async runtime")?
489        .block_on(serve_async());
490
491    Ok(())
492}
493
494async fn serve_async() {
495    let stdin = tokio::io::stdin();
496    let stdout = tokio::io::stdout();
497
498    let (service, socket) = LspService::new(Backend::new);
499
500    Server::new(stdin, stdout, socket).serve(service).await;
501}

503fn main() {
504    let _ = serve();
