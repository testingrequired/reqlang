260            }
261
262            if !secret_names.is_empty() {
263                ui.heading("Secrets");
264
265                ui.horizontal(|ui| {

351#[derive(Clone)]
352struct ExecutingRequestState {
353    reqfile: Box<TemplatedRequestFile>,
354    download: Arc<Mutex<Download>>,
355}
356
357impl ExecutingRequestState {
358    pub fn new(reqfile: Box<TemplatedRequestFile>) -> Self {
359        Self {
360            reqfile,
361            download: Arc::new(Mutex::new(Download::None)),
362        }
363    }
364
365    pub fn ui(
366        &mut self,
367        egui_ctx: &egui::Context,
368        _client_ctx: &ClientContext,
369    ) -> Result<StateTransition, &str> {
370        let mut next_state: StateTransition = StateTransition::None;
371
372        let request = &self.reqfile.as_ref().request;
373        let request_string = export(request, RequestFormat::HttpMessage);
374
375        egui::CentralPanel::default().show(egui_ctx, |ui| {
376            ui.heading("Request");
377
378            selectable_text(ui, &request_string);
379
380            if ui.button("Execute").clicked() {
381                eprintln!("CLICKED EXECUTE");
382                let verb = &request.verb;
383                let target = request.target.as_str();
384                let download_store = self.download.clone();
385                *download_store.lock().unwrap() = Download::InProgress;
386                let egui_ctx = egui_ctx.clone();
387
388                let request = match verb.to_string().as_str() {
389                    "GET" => ehttp::Request::get(target),
390                    "HEAD" => ehttp::Request::head(target),
391                    "POST" => ehttp::Request::post(
392                        target,
393                        request.body.clone().unwrap_or_default().as_bytes().to_vec(),
394                    ),
395                    _ => unreachable!(),
396                };
397
398                let streaming = true;
399
400                if streaming {
401                    eprintln!("STREAMING REQUEST");
402                    // The more complicated streaming API:
403                    ehttp::streaming::fetch(request, move |part| {
404                        egui_ctx.request_repaint(); // Wake up UI thread
405                        on_fetch_part(part, &mut download_store.lock().unwrap())
406                    });
407                } else {
408                    eprintln!("NON STREAMING REQUEST");
409                    // The simple non-streaming API:
410                    ehttp::fetch(request, move |response| {
411                        *download_store.lock().unwrap() = Download::Done(response);
412                        egui_ctx.request_repaint(); // Wake up UI thread
413                    });
414                }
415
416                ui.separator();
417            }
418
419            let download: &Download = &self.download.lock().unwrap();
420            match download {
421                Download::None => {}
422                Download::InProgress => {
423                    ui.label("Wait for itâ€¦");
424                }
425                Download::StreamingInProgress { body, .. } => {
426                    let num_bytes = body.len();
427                    if num_bytes < 1_000_000 {
428                        ui.label(format!("{:.1} kB", num_bytes as f32 / 1e3));
429                    } else {
430                        ui.label(format!("{:.1} MB", num_bytes as f32 / 1e6));
431                    }
432                }
433                Download::Done(response) => {
434                    if ui.button("Back").clicked() {
435                        next_state = StateTransition::Back;
436                        return;
437                    }
438
439                    ui.separator();
440
441                    match response {
442                        Err(err) => {
443                            ui.label(err);
444                        }
445                        Ok(response) => {
446                            response_ui(ui, response);
447                        }
448                    }
449                }
450            }
451        });
452
453        Ok(next_state)
454    }
455}

514enum StateTransition {
515    None,
516    Back,
517    New(ClientState),

520impl eframe::App for Client {
521    fn update(&mut self, egui_ctx: &egui::Context, _frame: &mut eframe::Frame) {
522        match self.states.last_mut() {
523            Some(latest_state) => {
524                let next_state = match &mut *latest_state {
525                    ClientState::LoadReqfile(state) => state.ui(egui_ctx, &mut self.context),
526                    ClientState::ViewReqfile(state) => state.ui(egui_ctx, &mut self.context),
527                    ClientState::InputParams(state) => state.ui(egui_ctx, &self.context),
528                    ClientState::ReviewRequest(state) => state.ui(egui_ctx, &self.context),
529                    ClientState::ExecutingRequest(state) => state.ui(egui_ctx, &self.context),
530                };
531
532                match next_state {
533                    Ok(next_state) => match next_state {
534                        StateTransition::None => {}
535                        StateTransition::Back => {
536                            self.states.pop();
537                        }
538                        StateTransition::New(next_state) => self.states.push(next_state),
539                    },
540                    Err(err) => panic!("{err}"),
541                };
542            }
543            None => panic!("Trying to pop non existant state from client"),
544        };
545    }
546}

548fn on_fetch_part(
549    part: Result<ehttp::streaming::Part, String>,
550    download_store: &mut Download,
551) -> ControlFlow<()> {
552    let part = match part {
553        Err(err) => {
554            *download_store = Download::Done(Result::Err(err));
555            return ControlFlow::Break(());
556        }
557        Ok(part) => part,
558    };
559
560    match part {
561        ehttp::streaming::Part::Response(response) => {
562            *download_store = Download::StreamingInProgress {
563                response,
564                body: Vec::new(),
565            };
566            ControlFlow::Continue(())
567        }
568        ehttp::streaming::Part::Chunk(chunk) => {
569            if let Download::StreamingInProgress { response, mut body } =
570                std::mem::replace(download_store, Download::None)
571            {
572                body.extend_from_slice(&chunk);
573
574                if chunk.is_empty() {
575                    // This was the last chunk.
576                    *download_store = Download::Done(Ok(response.complete(body)));
577                    ControlFlow::Break(())
578                } else {
579                    // More to come.
580                    *download_store = Download::StreamingInProgress { response, body };
581                    ControlFlow::Continue(())
582                }
583            } else {
584                ControlFlow::Break(()) // some data race - abort download.
585            }
586        }
587    }
588}

590fn response_ui(ui: &mut egui::Ui, response: &ehttp::Response) {
591    ui.monospace(format!("url:          {}", response.url));
592    ui.monospace(format!(
593        "status:       {} ({})",
594        response.status, response.status_text
595    ));
596    ui.monospace(format!(
597        "content-type: {}",
598        response.content_type().unwrap_or_default()
599    ));
600    ui.monospace(format!(
601        "size:         {:.1} kB",
602        response.bytes.len() as f32 / 1000.0
603    ));
604
605    ui.separator();
606
607    egui::ScrollArea::vertical().show(ui, |ui| {
608        egui::CollapsingHeader::new("Response headers")
609            .default_open(false)
610            .show(ui, |ui| {
611                egui::Grid::new("response_headers")
612                    .spacing(egui::vec2(ui.spacing().item_spacing.x * 2.0, 0.0))
613                    .show(ui, |ui| {
614                        for (k, v) in &response.headers {
615                            ui.label(k);
616                            ui.label(v);
617                            ui.end_row();
618                        }
619                    })
620            });
621
622        ui.separator();
623
624        if let Some(text) = response.text() {
625            let tooltip = "Click to copy the response body";
626            if ui.button("ðŸ“‹").on_hover_text(tooltip).clicked() {
627                ui.output_mut(|o| text.clone_into(&mut o.copied_text));
628            }
629            ui.separator();
630        }
631
632        if let Some(text) = response.text() {
633            selectable_text(ui, text);
634        } else {
635            ui.monospace("[binary]");
636        }
637    });
638}
