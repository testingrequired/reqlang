13/// Template a request file string into a [TemplatedRequestFile].
14pub fn template(
15    reqfile_string: &str,
16    env: Option<&str>,
17    prompts: &HashMap<String, String>,
18    secrets: &HashMap<String, String>,
19    provider_values: &HashMap<String, String>,
20) -> Result<TemplatedRequestFile, Vec<Spanned<ReqlangError>>> {
21    let ast = Ast::from(reqfile_string);
22    let parsed_reqfile = parse(&ast)?;
23
24    if let Some(env) = env {
25        match &parsed_reqfile.config {
26            Some((config, span)) => {
27                if config.envs().is_empty() {
28                    return Err(vec![(
29                        ResolverError::NoEnvironmentsDefined(env.to_string()).into(),
30                        span.clone(),
31                    )]);
32                }
33            }
34            None => {
35                return Err(vec![(
36                    ResolverError::NoEnvironmentsDefined(env.to_string()).into(),
37                    0..0,
38                )]);
39            }
40        }
41
42        if !parsed_reqfile.envs().contains(&env.to_string()) {
43            return Err(vec![(
44                ResolverError::InvalidEnvError(env.to_string()).into(),
45                parsed_reqfile.config.map(|x| x.1).unwrap_or_default(),
46            )]);
47        }
48    }
49
50    let mut templating_errors: Vec<Spanned<ReqlangError>> = vec![];
51
52    let reqfile: &ParsedRequestFile = &parsed_reqfile;
53
54    // Validate all required prompt values were passed
55    {
56        let missing_prompts_errs = {
57            parsed_reqfile
58                .required_prompts()
59                .into_iter()
60                .filter(|prompt| !prompts.contains_key(prompt))
61                .map(|prompt| ResolverError::PromptValueNotPassed(prompt.clone()).into())
62                .map(|err| (err, NO_SPAN))
63                .collect::<Vec<Spanned<ReqlangError>>>()
64        };
65
66        templating_errors.extend(missing_prompts_errs);
67    };
68
69    // Validate all required secret values were passed
70    {
71        let missing_secrets_errs = parsed_reqfile
72            .secrets()
73            .into_iter()
74            .filter(|secret| !secrets.contains_key(secret))
75            .map(|secret| ResolverError::SecretValueNotPassed(secret.clone()).into())
76            .map(|err| (err, NO_SPAN))
77            .collect::<Vec<Spanned<ReqlangError>>>();
78
79        templating_errors.extend(missing_secrets_errs);
80    };
81
82    if !templating_errors.is_empty() {
83        return Err(templating_errors);
84    }
85
86    // Replace template references with the resolved values
87    let templated_input = {
88        let mut input = reqfile_string.to_string();
89
90        // Gather list of template references along with each reference's type
91        //
92        // e.g. ("{{:var_name}}", ReferenceType::Variable("var_name"))
93        let template_refs_to_replace: Vec<(String, ReferenceType)> = reqfile
94            .refs
95            .iter()
96            .map(|(template_reference, _)| {
97                (format!("{template_reference}"), template_reference.clone())
98            })
99            .collect();
100
101        // If the environment is provided, use it to resolve template references
102        let vars = match env {
103            Some(env) => reqfile.env(env).unwrap_or_default(),
104            None => HashMap::new(),
105        };
106
107        let default_variable_values = parsed_reqfile.default_variable_values();
108        let default_prompt_values = parsed_reqfile.default_prompt_values();
109
110        // Replace template references with the resolved values
111        for (template_ref, ref_type) in &template_refs_to_replace {
112            let value = match ref_type {
113                ReferenceType::Variable(name) => {
114                    vars.get(name).or(default_variable_values.get(name))
115                }
116                ReferenceType::Prompt(name) => {
117                    prompts.get(name).or(default_prompt_values.get(name))
118                }
119                ReferenceType::Secret(name) => secrets.get(name),
120                ReferenceType::Provider(name) => provider_values.get(name),
121                _ => None,
122            };
123
124            // If reference can not be resolved, keep the template reference as is
125            if value.is_some() {
126                input = input.replace(template_ref, value.unwrap());
127            }
128        }
129
130        let mut compiler_env =
131            CompileTimeEnv::new(reqfile.vars(), reqfile.prompts(), reqfile.secrets(), vec![]);
132
133        let mut runtime_env = {
134            let var_values = match env {
135                Some(env) => reqfile
136                    .vars()
137                    .iter()
138                    .map(|x| {
139                        let config = parsed_reqfile.config.clone().unwrap().0.env(env).unwrap();
140                        let value = config.get(x).unwrap();
141
142                        value.clone()
143                    })
144                    .collect(),
145                None => vec![],
146            };
147
148            let prompt_values: Vec<Option<String>> = reqfile
149                .prompts()
150                .iter()
151                .map(|x| prompts.get(x).cloned())
152                .collect();
153
154            let secret_values: Vec<Option<String>> = reqfile
155                .secrets()
156                .iter()
157                .map(|x| secrets.get(x).cloned())
158                .collect();
159
160            RuntimeEnv {
161                vars: var_values.clone(),
162                prompts: prompt_values.iter().filter_map(|x| x.clone()).collect(),
163                secrets: secret_values.iter().filter_map(|x| x.clone()).collect(),
164                client_context: vec![],
165            }
166        };
167
168        // Set client context `@env` if an environment was provided
169        if let Some(env) = env {
170            let env_context_index = compiler_env.add_to_client_context("env");
171            runtime_env.add_to_client_context(env_context_index, Value::String(env.to_string()));
172        }
173
174        let mut vm = Vm::new();
175
176        let items = &reqfile.exprs.to_vec();
177        for (expr_ref, expr_span) in items {
178            match reqlang_expr::parser::parse(&format!("({expr_ref})")) {
179                Ok(expr) => match compile(&mut (expr, expr_span.clone()), &compiler_env) {
180                    Ok(bytecode) => {
181                        let result = vm.interpret(bytecode.into(), &compiler_env, &runtime_env);
182
183                        let replacement_string = match result {
184                            Ok(expr_value) => expr_value
185                                .get_string()
186                                .expect("should be string")
187                                .to_string(),
188                            Err(expr_errs) => {
189                                templating_errors.push((
190                                    ReqlangError::ResolverError(
191                                        ResolverError::ExpressionEvaluationError(
192                                            expr_ref.clone(),
193                                            format!("{expr_errs:#?}"),
194                                        ),
195                                    ),
196                                    expr_span.clone(),
197                                ));
198
199                                // In the case of an error, replacement string
200                                // is the original string
201                                expr_ref.clone()
202                            }
203                        };
204
205                        let x = format!("{{({expr_ref})}}");
206
207                        input = input.replace(&x, &replacement_string);
208                    }
209                    Err(expr_errs) => {
210                        templating_errors.push((
211                            ReqlangError::ResolverError(ResolverError::ExpressionEvaluationError(
212                                expr_ref.clone(),
213                                format!("{expr_errs:#?}"),
214                            )),
215                            expr_span.clone(),
216                        ));
217                    }
218                },
219                Err(expr_errs) => {
220                    templating_errors.push((
221                        ReqlangError::ResolverError(ResolverError::ExpressionEvaluationError(
222                            expr_ref.clone(),
223                            format!("{expr_errs:#?}"),
224                        )),
225                        expr_span.clone(),
226                    ));
227                }
228            }
229        }
230
231        if !templating_errors.is_empty() {
232            return Err(templating_errors.clone());
233        }
234
235        input
236    };
237
238    let ast = Ast::from(&templated_input);
239    let request = ast.request().cloned().expect("should have a request");
240    let response = ast.response().cloned();
241
242    // Parse the templated request
243    let request = {
244        let (request, request_span) = request;
245        parse_request(&(request, request_span.clone())).unwrap().0
246    };
247
248    // Parse the templated response
249    let response = parse_response(&response).map(|x| x.unwrap().0);
250
251    Ok(TemplatedRequestFile { request, response })
252}
