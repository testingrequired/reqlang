5/// Abstract syntax tree for a request file
6#[derive(Debug, PartialEq, Serialize, Deserialize)]
7pub struct Ast(Vec<Spanned<AstNode>>);
8
9impl Ast {
10    /// Create an [AST](Ast) from a collection of [nodes](AstNode)
11    pub fn new(nodes: Vec<Spanned<AstNode>>) -> Self {
12        Self(nodes)
13    }
14
15    /// Create an [AST](Ast) by parsing a string in to a collection of [nodes](AstNode)
16    pub fn from(input: impl AsRef<str>) -> Self {
17        let mut nodes: Vec<Spanned<AstNode>> = vec![];
18
19        for (text, span) in extract_codeblocks(&input, "%request").iter() {
20            nodes.push((AstNode::RequestBlock(text.clone()), span.clone()));
21        }
22
23        for (text, span) in extract_codeblocks(&input, "%config").iter() {
24            nodes.push((AstNode::ConfigBlock(text.clone()), span.clone()));
25        }
26
27        for (text, span) in extract_codeblocks(&input, "%response").iter() {
28            nodes.push((AstNode::ResponseBlock(text.clone()), span.clone()));
29        }
30
31        // Sort AST nodes by their positions
32        nodes.sort_by(|a, b| a.1.start.cmp(&b.1.start));
33
34        let mut index = 0usize;
35
36        // Find comment nodes
37        for (_, node_span) in nodes.clone().iter() {
38            let start = node_span.start;
39
40            if index < start {
41                let new_span = index..start;
42                let comment = input.as_ref()[new_span.clone()].to_string();
43                nodes.push((AstNode::Comment(comment), new_span.clone()));
44                index = node_span.end;
45            }
46        }
47
48        // Sort AST nodes by their positions
49        nodes.sort_by(|a, b| a.1.start.cmp(&b.1.start));
50
51        Self::new(nodes)
52    }
53
54    /// Iterate over the [nodes](AstNode)
55    pub fn iter(&self) -> impl Iterator<Item = &Spanned<AstNode>> {
56        self.0.iter()
57    }
58
59    /// Get the [AstNode::ConfigBlock], if present
60    pub fn config(&self) -> Option<&Spanned<String>> {
61        self.iter().find_map(|(node, _)| match &node {
62            AstNode::ConfigBlock(config) => Some(config),
63            _ => None,
64        })
65    }
66
67    /// Get the [AstNode::RequestBlock]
68    pub fn request(&self) -> Option<&Spanned<String>> {
69        self.iter().find_map(|(node, _)| match &node {
70            AstNode::RequestBlock(request) => Some(request),
71            _ => None,
72        })
73    }
74
75    /// Get the [AstNode::ResponseBlock], if present
76    pub fn response(&self) -> Option<&Spanned<String>> {
77        self.iter().find_map(|(node, _)| match &node {
78            AstNode::ResponseBlock(response) => Some(response),
79            _ => None,
80        })
81    }
82
83    /// Get all [AstNode::Comment]
84    pub fn comments(&self) -> Vec<Spanned<String>> {
85        let mut comments = vec![];
86
87        for node in self.iter() {
88            if let AstNode::Comment(comment) = &node.0 {
89                comments.push((comment.clone(), node.1.clone()));
90            }
91        }
92
93        comments
94    }
95}

97#[derive(Debug, PartialEq, Clone, Serialize, Deserialize)]
98pub enum AstNode {
99    /// Any text that isn't a [AstNode::RequestBlock], [AstNode::ResponseBlock], or [AstNode::ConfigBlock].
100    Comment(String),
101    /// A code block delimited configuration
102    ConfigBlock(Spanned<String>),
103    /// A code block delimited request
104    RequestBlock(Spanned<String>),
105    /// A code block delimited response
106    ResponseBlock(Spanned<String>),
107}
