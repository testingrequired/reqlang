1use crate::types::ReferenceType;
2use serde::{Deserialize, Serialize};
3use thiserror::Error;
4
5/// Common error for parsing and templating request files
6#[derive(Debug, Clone, Error, PartialEq, Serialize, Deserialize)]
7pub enum ReqlangError {
8    #[error("ParseError: {0}")]
9    ParseError(ParseError),
10    #[error("ResolverError: {0}")]
11    ResolverError(ResolverError),
12}
13
14#[derive(Debug, Clone, Error, PartialEq, Serialize, Deserialize)]
15pub enum ParseError {
16    #[error("Request file requires a request be defined")]
17    MissingRequest,
18    #[error("Request is invalid: {message}")]
19    InvalidRequestError { message: String },
20    #[error("Config is invalid: {message}")]
21    InvalidConfigError { message: String },
22    #[error("Undefined template reference: {0}")]
23    UndefinedReferenceError(ReferenceType),
24    #[error(
25        "Value was declared but not used. Try adding the template reference {0} to the request or response."
26    )]
27    UnusedValueError(ReferenceType),
28    #[error(
29        "This request header is calculated at request time and can not be specified by user: {0}"
30    )]
31    ForbiddenRequestHeaderNameError(String),
32    #[error("Variable '{0}' is undefined in the environment '{1}'")]
33    VariableUndefinedInEnvironment(String, String),
34    #[error("Variable '{0}' is not defined in any environment or no environments are defined")]
35    VariableNotDefinedInAnyEnvironment(String),
36}
37
38#[derive(Debug, Clone, Error, PartialEq, Serialize, Deserialize)]
39pub enum ResolverError {
40    #[error("'{0}' is not a defined environment in the request file")]
41    InvalidEnvError(String),
42    #[error(
43        "Trying to resolve the environment '{0}' but no environments are defined in the request file"
44    )]
45    NoEnvironmentsDefined(String),
46    #[error("Prompt required but not passed: {0}")]
47    PromptValueNotPassed(String),
48    #[error("Secret required but not passed: {0}")]
49    SecretValueNotPassed(String),
50    #[error("There was an error evaluating the expression: '{0}'; Error: {1}")]
51    ExpressionEvaluationError(String, String),
52}
53
54macro_rules! impl_from_error {
55    ($($error:tt),+) => {$(
56        impl From<$error> for ReqlangError {
57            fn from(e: $error) -> Self {
58                ReqlangError::$error(e)
59            }
60        }
61    )+};
62}

64impl_from_error!(ParseError, ResolverError);
